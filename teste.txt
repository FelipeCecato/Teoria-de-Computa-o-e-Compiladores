{ Outro programa de exemplo em PL/0 }


CONST
  max = 3, min = 1;

VAR
  i, sum, temp;

PROCEDURE CalculateSum;
BEGIN
  sum := 0;
  i := min;
  WHILE i <= max DO 
  BEGIN
    sum := sum + i;
    i := i + 1;
  END
END;

BEGIN
  sum := 0;
  temp := max - min;
  CALL CalculateSum;
  IF sum >= temp THEN
    BEGIN
      i := 10;
      WHILE i > 0 DO
      BEGIN
        sum := sum + i;
        i := i - 1
      END
    END
END.
